**tombstone.ts** — Create new file:
```typescript
import type Database from 'better-sqlite3'

/**
 * Purge soft-deleted records older than 90 days from nodes, documents, and bookmarks.
 * Called once at server startup after migrations.
 *
 * Per ARCHITECTURE.md Section 3: tombstones are retained for 90 days to allow
 * offline devices to sync deletions, then permanently removed.
 */
export function purgeTombstones(sqlite: InstanceType<typeof Database>): void {
    const cutoff = Date.now() - 90 * 24 * 60 * 60 * 1000
    const tables = ['nodes', 'documents', 'bookmarks'] as const
    for (const table of tables) {
        const result = sqlite
            .prepare(`DELETE FROM ${table} WHERE deleted_at IS NOT NULL AND deleted_at < ?`)
            .run(cutoff)
        if (result.changes > 0) {
            console.log(`Purged ${result.changes} tombstones from ${table}`)
        }
    }
}
```

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest'
import Database from 'better-sqlite3'
import { runMigrations } from './db/migrate.js'
import { purgeTombstones } from './tombstone.js'

describe('purgeTombstones', () => {
    let sqlite: InstanceType<typeof Database>

    beforeEach(async () => {
        sqlite = new Database(':memory:')
        sqlite.pragma('journal_mode = WAL')
        sqlite.pragma('foreign_keys = ON')
        await runMigrations(sqlite)
    })

    afterEach(() => {
        sqlite.close()
    })

    it('deletes nodes with deleted_at older than 90 days', () => {
        const oldTimestamp = Date.now() - 91 * 24 * 60 * 60 * 1000
        const recentTimestamp = Date.now() - 10 * 24 * 60 * 60 * 1000

        // Insert a user first (foreign key)
        sqlite.prepare(
            'INSERT INTO users (id, google_sub, email, name, picture, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?)'
        ).run('u1', 'sub1', 'a@b.com', 'Test', '', Date.now(), Date.now())

        // Old tombstone (should be purged)
        sqlite.prepare(
            'INSERT INTO nodes (id, content, content_hlc, note, note_hlc, parent_id_hlc, sort_order, sort_order_hlc, completed_hlc, color_hlc, collapsed_hlc, deleted_at, deleted_hlc, device_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)'
        ).run('n1', '', '', '', '', '', 'a0', '', '', '', '', oldTimestamp, '', 'd1', Date.now(), Date.now())

        // Recent tombstone (should NOT be purged)
        sqlite.prepare(
            'INSERT INTO nodes (id, content, content_hlc, note, note_hlc, parent_id_hlc, sort_order, sort_order_hlc, completed_hlc, color_hlc, collapsed_hlc, deleted_at, deleted_hlc, device_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)'
        ).run('n2', '', '', '', '', '', 'a0', '', '', '', '', recentTimestamp, '', 'd1', Date.now(), Date.now())

        // Active node (should NOT be purged)
        sqlite.prepare(
            'INSERT INTO nodes (id, content, content_hlc, note, note_hlc, parent_id_hlc, sort_order, sort_order_hlc, completed_hlc, color_hlc, collapsed_hlc, deleted_hlc, device_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)'
        ).run('n3', 'active', '', '', '', '', 'a0', '', '', '', '', '', 'd1', Date.now(), Date.now())

        purgeTombstones(sqlite)

        const remaining = sqlite.prepare('SELECT id FROM nodes').all() as { id: string }[]
        expect(remaining.map(r => r.id).sort()).toEqual(['n2', 'n3'])
    })

    it('deletes documents with deleted_at older than 90 days', () => {
        const oldTimestamp = Date.now() - 91 * 24 * 60 * 60 * 1000

        sqlite.prepare(
            'INSERT INTO users (id, google_sub, email, name, picture, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?)'
        ).run('u1', 'sub1', 'a@b.com', 'Test', '', Date.now(), Date.now())

        // Insert old tombstone document — use the exact column names from the migration
        sqlite.prepare(
            'INSERT INTO documents (id, user_id, title, title_hlc, sort_order, sort_order_hlc, deleted_at, deleted_hlc, device_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)'
        ).run('d1', 'u1', 'old doc', '', 'a0', '', oldTimestamp, '', 'dev1', Date.now(), Date.now())

        purgeTombstones(sqlite)

        const remaining = sqlite.prepare('SELECT id FROM documents').all()
        expect(remaining).toHaveLength(0)
    })

    it('deletes bookmarks with deleted_at older than 90 days', () => {
        const oldTimestamp = Date.now() - 91 * 24 * 60 * 60 * 1000

        sqlite.prepare(
            'INSERT INTO users (id, google_sub, email, name, picture, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?)'
        ).run('u1', 'sub1', 'a@b.com', 'Test', '', Date.now(), Date.now())

        // Insert old tombstone bookmark
        sqlite.prepare(
            'INSERT INTO bookmarks (id, user_id, node_id, node_id_hlc, sort_order, sort_order_hlc, deleted_at, deleted_hlc, device_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)'
        ).run('b1', 'u1', 'n1', '', 'a0', '', oldTimestamp, '', 'dev1', Date.now(), Date.now())

        purgeTombstones(sqlite)

        const remaining = sqlite.prepare('SELECT id FROM bookmarks').all()
        expect(remaining).toHaveLength(0)
    })

    it('does not purge recent tombstones', () => {
        const recentTimestamp = Date.now() - 30 * 24 * 60 * 60 * 1000

        sqlite.prepare(
            'INSERT INTO users (id, google_sub, email, name, picture, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?)'
        ).run('u1', 'sub1', 'a@b.com', 'Test', '', Date.now(), Date.now())

        sqlite.prepare(
            'INSERT INTO nodes (id, content, content_hlc, note, note_hlc, parent_id_hlc, sort_order, sort_order_hlc, completed_hlc, color_hlc, collapsed_hlc, deleted_at, deleted_hlc, device_id, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)'
        ).run('n1', '', '', '', '', '', 'a0', '', '', '', '', recentTimestamp, '', 'd1', Date.now(), Date.now())

        purgeTombstones(sqlite)

        const remaining = sqlite.prepare('SELECT id FROM nodes').all()
        expect(remaining).toHaveLength(1)
    })
})
```
