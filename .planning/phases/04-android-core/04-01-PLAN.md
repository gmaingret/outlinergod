**Part A ŌĆö Safety net in NodeEditorViewModel.loadDocument():**

In the `collect` block of `loadDocument()` (lines 57-68), before calling `mapToFlatList`, add an isEmpty guard:

```kotlin
fun loadDocument(documentId: String) = intent {
    reduce { state.copy(documentId = documentId, status = NodeEditorStatus.Loading) }
    nodeDao.getNodesByDocument(documentId).collect { nodes ->
        if (nodes.isEmpty()) {
            // Safety net: create a root node so the editor is never blank.
            // Primary fix is in DocumentListViewModel.createDocument(); this
            // guard handles documents imported or synced without a root node.
            val deviceId = authRepository.getDeviceId().first()
            val now = System.currentTimeMillis()
            val hlc = hlcClock.generate(deviceId)
            val userId = authRepository.getAccessToken().filterNotNull().first()
            val rootNode = NodeEntity(
                id = java.util.UUID.randomUUID().toString(),
                documentId = documentId,
                userId = userId,
                content = "",
                contentHlc = hlc,
                note = "",
                noteHlc = "",
                parentId = documentId,
                parentIdHlc = hlc,
                sortOrder = "a0",
                sortOrderHlc = hlc,
                deviceId = deviceId,
                createdAt = now,
                updatedAt = now,
            )
            nodeDao.insertNode(rootNode)
            // Flow will re-emit with the new node; return so we don't set
            // Success state with an empty flatNodes list.
            return@collect
        }
        val flatNodes = mapToFlatList(nodes, documentId)
        reduce {
            state.copy(
                status = NodeEditorStatus.Success,
                flatNodes = flatNodes,
            )
        }
    }
}
```
