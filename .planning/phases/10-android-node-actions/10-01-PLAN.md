    ```kotlin
    fun restoreNode(nodeId: String) = intent {
        val documentId = state.documentId
        val entity = nodeDao.getNodesByDocumentSync(documentId)
            .firstOrNull { it.id == nodeId }
        if (entity != null) {
            // Node still exists but is soft-deleted -- just clear deletedAt won't work
            // because getNodesByDocumentSync filters WHERE deleted_at IS NULL.
            // We need to get it differently. Use nodeDao.getNodeById() instead.
        }
        // Actually: softDeleteNode sets deleted_at on the node. getNodesByDocumentSync
        // filters by deleted_at IS NULL, so we can't retrieve the deleted node that way.
        // Instead, use getNodeById which has no deleted_at filter.
        val node = nodeDao.getNodeById(nodeId).first() ?: return@intent
        val deviceId = authRepository.getDeviceId().first()
        val hlc = hlcClock.generate(deviceId)
        val now = System.currentTimeMillis()
        nodeDao.updateNode(
            node.copy(
                deletedAt = null,
                deletedHlc = hlc,
                updatedAt = now,
                deviceId = deviceId,
            )
        )
        resetInactivityTimer()
    }
    ```
