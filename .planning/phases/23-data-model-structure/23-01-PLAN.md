       ```kotlin
       val MIGRATION_2_3 = object : Migration(2, 3) {
           override fun migrate(db: SupportSQLiteDatabase) {
               db.execSQL("ALTER TABLE nodes ADD COLUMN attachment_url TEXT")
               db.execSQL("ALTER TABLE nodes ADD COLUMN attachment_mime TEXT")
               // Format: ATTACH|mime|filename|url (4 parts, 3 pipes)
               // SUBSTR(content,8) = "mime|filename|url"
               // INSTR(SUBSTR(content,8),'|') = position of first pipe after "ATTACH|" offset = end of mime
               // mime = SUBSTR(content, 8, first_pipe_pos - 1)
               // skip to after filename: 8 + first_pipe_pos = start of filename
               // INSTR of second segment = end of filename
               // url starts after third pipe
               db.execSQL("""
                   UPDATE nodes SET
                     attachment_mime = SUBSTR(content, 8, INSTR(SUBSTR(content, 8), '|') - 1),
                     attachment_url  = SUBSTR(content,
                         8 + INSTR(SUBSTR(content, 8), '|')
                           + INSTR(SUBSTR(content, 8 + INSTR(SUBSTR(content, 8), '|')), '|')),
                     content = ''
                   WHERE content LIKE 'ATTACH|%'
               """.trimIndent())
           }
       }
       ```
